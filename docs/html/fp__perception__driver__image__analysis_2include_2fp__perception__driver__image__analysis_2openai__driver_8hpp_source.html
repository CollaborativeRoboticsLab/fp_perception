163 static std::string base64_encode(const unsigned char* data, size_t len)
164 {
165 static const char* kTable = "ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz0123456789+/";
166 std::string out;
167 out.reserve(((len + 2) / 3) * 4);
168
169 for (size_t i = 0; i < len; i += 3)
170 {
171 const unsigned int octet_a = i < len ? data[i] : 0;
172 const unsigned int octet_b = (i + 1) < len ? data[i + 1] : 0;
173 const unsigned int octet_c = (i + 2) < len ? data[i + 2] : 0;
174
175 const unsigned int triple = (octet_a << 16) | (octet_b << 8) | octet_c;
176
177 out.push_back(kTable[(triple >> 18) & 0x3F]);
178 out.push_back(kTable[(triple >> 12) & 0x3F]);
179 out.push_back((i + 1) < len ? kTable[(triple >> 6) & 0x3F] : '=');
180 out.push_back((i + 2) < len ? kTable[triple & 0x3F] : '=');
181 }
182
183 return out;
184 }
