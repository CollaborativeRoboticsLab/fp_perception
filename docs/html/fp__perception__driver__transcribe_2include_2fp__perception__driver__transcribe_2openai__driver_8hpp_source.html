104 RCLCPP_INFO(node_->get_logger(),
105 "Starting OpenAI transcription request: samples=%zu sample_rate=%d channels=%d duration=%.3f seconds",
106 new_audio.samples.size(), new_audio.sample_rate, new_audio.channels, audio_seconds);
107
108 // create fp_perception::RESTRequest object
109 fp_perception::RESTRequest request;

111 request.file_stream = encodeWavToBytes(new_audio);
112 request.options.clear();
113
114 fp_perception::RESTOption model_option1;
115 model_option1.key = "model";
116 model_option1.value = model_name_;
117 model_option1.type = fp_perception::RESTOptionType::STRING;
